package com.rediff.pageobject;

import java.awt.Desktop.Action;

import javax.xml.xpath.XPath;

import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.interactions.Actions;
import org.openqa.selenium.support.FindBy;
import org.openqa.selenium.support.PageFactory;

import com.rediff.basepage.H_Basepage;

import net.bytebuddy.asm.Advice.This;

public class PO_applicationpage extends H_Basepage{
	@FindBy(xpath = "//*[@id=\"main-section\"]/nav")
	private WebElement services;
	@FindBy(xpath = "//*[@id=\"navbarTogglerDemo03\"]/ul/li[2]/a")
	private WebElement shopify;
	@FindBy (xpath="//*[@id=\"navbarTogglerDemo03\"]/ul/li[3]/a")
	private WebElement blog;
	@FindBy (xpath = "//*[@id=\"navbarTogglerDemo03\"]/ul/li[4]/a")
	private WebElement careers;
	@FindBy (xpath = "//*[@id=\"navbarTogglerDemo03\"]/ul/li[5]/a")
	private WebElement casestudies;
	@FindBy(xpath = "//*[@id=\"navbarTogglerDemo03\"]/ul/li[6]/a")
	private WebElement aboutus;
	@FindBy(xpath = "//*[@id=\"navbarTogglerDemo03\"]/ul/li[7]/a")
	private WebElement contactus;
	@FindBy(xpath = "//*[@id=\"navbarTogglerDemo03\"]/div/button[1]")
	private WebElement EN;
	@FindBy(linkText = "Let's Speak")
	private WebElement letsspeak;
	@FindBy (name = "name")
	private WebElement name;
	@FindBy(name = "email")
	private WebElement email;
	@FindBy(name = "phone")
	private WebElement phone;
	@FindBy(name = "resume")
	private WebElement resume;
	@FindBy(name = "description")
	private WebElement description;
	@FindBy(xpath = "//*[@id=\"contact-form\"]/div/div[7]/div/button[1]")
	private WebElement apply;
	
	public PO_applicationpage(WebDriver driver) {
		this.driver=driver;
		PageFactory.initElements(driver,this);
	}
	
	public void services() {
		Actions act=new Actions(driver);
		act.moveToElement(services).perform();
	}
	public void clickshopify() {
		shopify.click();
		
	}
	public void clickblog() {
		blog.click();
		
	}
	public void careers() {
		careers.click();
		
	}
	public void casestudies() {
		casestudies.click();
	}
	public void aboutus() {
		aboutus.click();
		
	}
	public void contactus() {
		contactus.click();
		
	}
	public void clickEN() {
		EN.click();
		
	}
	public void entername() {
		name.sendKeys(prop.getProperty("Name"));
		
	}
	public void entermailid() {
		email.sendKeys(prop.getProperty("Email"));
		
	}
	private void enterNO() {
		phone.sendKeys(prop.getProperty("Phone"));
		
	}
	private void description() {
		description.sendKeys(prop.getProperty("Description"));
		
	}
		
	public void clickapplybutton() {
		services();
		clickshopify();
		clickblog();
		careers();
		casestudies();
		aboutus();
		contactus();
		clickEN();
		entername();
		entermailid();
		enterNO();
		description();
		
	}
		
	}
